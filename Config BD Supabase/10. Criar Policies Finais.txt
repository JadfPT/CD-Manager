-- cd_albums
create policy "cd_albums_select_authenticated"
on public.cd_albums
for select
to authenticated
using (not is_archived);

create policy "cd_albums_insert_admin"
on public.cd_albums
for insert
to authenticated
with check (public.current_user_is_admin());

create policy "cd_albums_update_admin"
on public.cd_albums
for update
to authenticated
using (public.current_user_is_admin())
with check (public.current_user_is_admin());
